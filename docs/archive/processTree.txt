#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    int n;
    printf("Enter fork(): ");
    scanf("%d", &n);

    FILE *out = fopen("pids.txt", "a");  // Change "w" to "a" to append
    if (out == NULL) {
        perror("Error opening file");
        return 1;
    }

    for (int i = 0; i < n; i++) {
        pid_t pid = fork();

        if (pid == 0) {  // Child process
            fprintf(out, "PID: %d, PPID: %d\n", getpid(), getppid());
            fflush(out);  // Ensure immediate write
            fclose(out);  // Close file before exiting
            exit(0);
        } 
        else if (pid > 0) {  // Parent process
            waitpid(pid, NULL, 0);  // Wait for child before continuing
        } 
        else {  // Fork failure
            perror("Fork failed");
            fclose(out);
            return 1;
        }
    }

    fclose(out);
    return 0;
}
